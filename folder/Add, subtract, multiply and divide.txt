<div>#include &lt;stdio.h&gt;</div><div>int main()</div><div>{</div><div>&nbsp; &nbsp;int first, second, add, subtract, multiply;</div><div>&nbsp; &nbsp;float divide;</div><div>&nbsp;</div><div>&nbsp; &nbsp;printf("Enter two integers\n");</div><div>&nbsp; &nbsp;scanf("%d%d", &amp;first, &amp;second);</div><div>&nbsp;</div><div>&nbsp; &nbsp;add &nbsp; &nbsp; &nbsp; &nbsp;= first + second;</div><div>&nbsp; &nbsp;subtract = first - second;</div><div>&nbsp; &nbsp;multiply = first * second;</div><div>&nbsp; &nbsp;divide &nbsp; &nbsp; = first / (float)second; &nbsp; //typecasting</div><div>&nbsp;</div><div>&nbsp; &nbsp;printf("Sum = %d\n",add);</div><div>&nbsp; &nbsp;printf("Difference = %d\n",subtract);</div><div>&nbsp; &nbsp;printf("Multiplication = %d\n",multiply);</div><div>&nbsp; &nbsp;printf("Division = %.2f\n",divide);</div><div>&nbsp;</div><div>&nbsp; &nbsp;return 0;</div><div>}</div><div><br></div><div><br></div><div>Description :</div><div><br></div><div>In c language when we divide two integers we get integer result for example 5/2 evaluates to 2.
